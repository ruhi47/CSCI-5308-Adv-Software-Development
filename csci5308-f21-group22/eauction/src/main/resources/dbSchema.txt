CREATE TABLE `USERS` (
  `USER_ID` varchar(10) NOT NULL,
  `FIRST_NAME` varchar(50) NOT NULL,
  `LAST_NAME` varchar(50) NOT NULL,
  `CREATED_DATE` datetime NOT NULL,
  `EMAIL` varchar(50) NOT NULL,
  `PASSWORD` varchar(100) NOT NULL,
  `DATE_OF_BIRTH` date DEFAULT NULL,
  `STATUS` varchar(50) NOT NULL,
  `ROLE` varchar(50) NOT NULL,
  `LAST_LOGGED_IN_DATE` datetime DEFAULT NULL,
  `CREDITS` decimal(10,0) DEFAULT NULL,
  PRIMARY KEY (`USER_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

  
  
CREATE TABLE `ADDRESS` (
  `ADDRESS_ID` int(11) NOT NULL AUTO_INCREMENT,
  `USER_ID` varchar(10) NOT NULL,
  `STREET_1` varchar(45) DEFAULT NULL,
  `STREET_2` varchar(45) DEFAULT NULL,
  `CITY` varchar(45) NOT NULL,
  `PROVINCE` varchar(45) NOT NULL,
  `COUNTRY` varchar(45) NOT NULL,
  `POSTAL_CODE` varchar(10) NOT NULL,
  PRIMARY KEY (`ADDRESS_ID`),
  KEY `USER_ID` (`USER_ID`),
  CONSTRAINT `USER_ID` FOREIGN KEY (`USER_ID`) REFERENCES `USERS` (`USER_ID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=7 DEFAULT CHARSET=utf8mb4;

CREATE TABLE `ITEMS` (
  `ITEM_ID` int(10) NOT NULL AUTO_INCREMENT,
  `ITEM_NAME` varchar(50) NOT NULL,
  `BRAND_NAME` varchar(50) DEFAULT NULL,
  `ITEM_DESCRIPTION` varchar(50) NOT NULL,
  `ITEM_PURCHASE_DATE` date DEFAULT NULL,
  `ITEM_EXPIRY_DATE` date DEFAULT NULL,
  `ITEM_MIN_PRICE` float NOT NULL,
  `ITEM_STATUS` varchar(50) NOT NULL,
  `ITEM_CATEGORY` varchar(50) NOT NULL,
  `USER_ID` varchar(10) NOT NULL,
  `IMAGE_URL` varchar(50) NOT NULL,
  PRIMARY KEY (`ITEM_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=14 DEFAULT CHARSET=utf8mb4;
